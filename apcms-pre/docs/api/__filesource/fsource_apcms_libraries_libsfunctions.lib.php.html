  320. function apcms_GetHTML($url) { 
  321.     global $errno, $errstr, $apcms, $db;
  322.     $url=trim($url);
  323.     if(!isset($url)) return ''; 
  324.     $status['html'] = '';
  325.     $urlArray = parse_url($url); 
  326.     if (!isset($urlArray['port']))  
  327.         $urlArray['port'] = "80"; 
  328.     if (!isset($urlArray['path']))  
  329.         $urlArray['path'] = "/";
  330.     if (ereg("\?", $url)) {
  331.         $parts = explode("?", $url);
  332.         $urlArray['path'] .= "?".$parts[1];
  333.     }
  334.     $sock = fsockopen($urlArray['host'], $urlArray['port'], $errno, $errstr, 2); 
  335.     if (!$sock) { 
  336.         $status['code'] = "Dead"; 
  337.         $status['contentType'] = ''; 
  338.         $status['html'] = ''; 
  339.         $status['msg'] = ''; 
  340.     } else { 
  341.         $dump = "GET ".$urlArray['path']." HTTP/1.1\r\n"; 
  342.         $dump .= "User-Agent: APCms/".$apcms['version']." (title: ".$apcms['title'].")\r\n";
  343.         $dump .= "Referer: ".$apcms['baseURL']."\r\n";
  344.         $dump .= "Host: ".$urlArray['host']."\r\n\r\n";
  345.         fputs($sock, $dump); 
  346.         socket_set_timeout($sock, 2); 
  347.         while ($str = fgets($sock, 1024)) { 
  348.             $status['html'] .= $str; 
  349.             if (eregi("^http/[0-9]+.[0-9]+ ([0-9]{3}) [a-z ]*", $str)) { 
  350.                 $status['code'] = trim(eregi_replace("^http/[0-9]+.[0-9]+ ([0-9]{3})[a-z ]*", "\\1", $str)); 
  351.             } 
  352.             if (eregi("^Content-Type: ", $str)) { 
  353.                 $status['contentType'] = trim(eregi_replace("^Content-Type: ", "", $str)); 
  354.             }
  355.         } 
  356.     } 
  357.     fclose($sock); 
  358.     if (!isset($status['code']) || trim($status['code'])=="") 
  359.         $status['code'] = "Dead"; 
  360.     if (!isset($status['contentType']) || trim($status['contentType']) == "") 
  361.         $status['contentType'] = ""; 
  362.     return $status; 
  363. }

  429. function apcms_ChmodDirectory($sourcedir, $mode=777) {
  430.

  432.     $filemode = octdec($filemode);
  433.     
  434.     if (!is_dir($sourcedir)) {
  435.         return chmod($sourcedir, $filemode);
  436.     }
  437.     
  438.     $filearray = array();
  439.     $sourcedir .= "/";
  440.     $handle = @opendir($sourcedir);
  441.     while ($eintrag = @readdir($handle)) {
  442.         $target = $sourcedir.$eintrag;
  443.         if (is_dir($target) && $eintrag != "." && $eintrag != "..") {
  444.             @chmod($target, $filemode);
  445.             $slashpos = strrpos($target, "/");
  446.             if ($slashpos) {
  447.                 $chmoddir = substr($target, 0, $slashpos);
  448.                 @chmod($chmoddir, $filemode);
  449.             }
  450.             $filearray[$target] = apcms_ChmodDirectory($target, $mode);
  451.             @chmod($target, $filemode);
  452.             $slashpos = strrpos($target, "/");
  453.             if ($slashpos) {

  455.                 @chmod($chmoddir, $filemode);
  456.             }
  457.         } elseif ($eintrag != "." && $eintrag != "..") {
  458.             $filearray[] = $eintrag;
  459.             @chmod($target, $filemode);

  461.             if ($slashpos) {

  465.         }
  466.     }
  467.     @closedir($handle);
  468.     return $filearray;
