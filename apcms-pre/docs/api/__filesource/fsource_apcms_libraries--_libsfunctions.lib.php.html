  123. function apcms_GenRandomString($pwdLen=32,$usables=PWD_ALLOW_ALL) {
  124.     $pwdSource = "";
  125.     $STRING = "";
  126.     if ( $usables & ( 1 << 0 ))     $pwdSource .= "1234567890";

  128.     if ( $usables & ( 1 << 2 ))     $pwdSource .= "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
  129.     srand ((double) microtime() * 1000000);
  130.     while ( $pwdLen ) {
  131.         srand ((double) microtime() * 1000000);
  132.         $STRING .= substr( $pwdSource, rand( 0, strlen( $pwdSource )), 1);
  133.         $pwdLen--;
  134.     }
  135.     if (strlen($STRING) <= ($pwdLen-1)) {
  136.         $dif = $pwdLen - strlen($STRING);
  137.         srand ((double) microtime() * 1000000);
  138.         while ( $dif ) {
  139.             srand ((double) microtime() * 1000000);
  140.             $STRING .= substr( $pwdSource, rand( 0, strlen( $pwdSource )), 1);
  141.             $dif--;
  142.         }
  143.     }
  144.     if (strlen($STRING) <= ($pwdLen-1)) {
  145.         $dif = $pwdLen - strlen($STRING);

  315.  * Fetches a complete Webpage from a given URI

  322. function apcms_GetHTML($url) { 
  323.     global $errno, $errstr, $apcms, $db;
  324.     $url=trim($url);
  325.     if(!isset($url)) return ''; 
  326.     $status['html'] = '';
  327.     $urlArray = parse_url($url); 
  328.     if (!isset($urlArray['port']))  
  329.         $urlArray['port'] = "80"; 
  330.     if (!isset($urlArray['path']))  
  331.         $urlArray['path'] = "/";
  332.     if (ereg("\?", $url)) {
  333.         $parts = explode("?", $url);
  334.         $urlArray['path'] .= "?".$parts[1];
  335.     }
  336.     $sock = fsockopen($urlArray['host'], $urlArray['port'], $errno, $errstr, 2); 
  337.     if (!$sock) { 
  338.         $status['code'] = "Dead"; 
  339.         $status['contentType'] = ''; 
  340.         $status['html'] = ''; 
  341.         $status['msg'] = ''; 
  342.     } else { 
  343.         $dump = "GET ".$urlArray['path']." HTTP/1.1\r\n"; 
  344.         $dump .= "User-Agent: APCms/".$apcms['version']." (title: ".$apcms['title'].")\r\n";
  345.         $dump .= "Referer: ".$apcms['baseURL']."\r\n";
  346.         $dump .= "Host: ".$urlArray['host']."\r\n\r\n";
  347.         fputs($sock, $dump); 
  348.         socket_set_timeout($sock, 2); 
  349.         while ($str = fgets($sock, 1024)) { 
  350.             $status['html'] .= $str; 
  351.             if (eregi("^http/[0-9]+.[0-9]+ ([0-9]{3}) [a-z ]*", $str)) { 
  352.                 $status['code'] = trim(eregi_replace("^http/[0-9]+.[0-9]+ ([0-9]{3})[a-z ]*", "\\1", $str)); 
  353.             } 
  354.             if (eregi("^Content-Type: ", $str)) { 
  355.                 $status['contentType'] = trim(eregi_replace("^Content-Type: ", "", $str)); 
  356.             }
  357.         } 
  358.     } 
  359.     fclose($sock); 
  360.     if (!isset($status['code']) || trim($status['code'])=="") 
  361.         $status['code'] = "Dead"; 
  362.     if (!isset($status['contentType']) || trim($status['contentType']) == "") 
  363.         $status['contentType'] = ""; 
  364.     return $status; 
  365. }

  379. function apcms_DeleteDirectory($sourcedir) {
  380.     
  381.     if (!is_dir($sourcedir)) {
  382.         return unlink($sourcedir);
  383.     }
  384.     
  385.     $filearray = array();
  386.     $sourcedir .= "/";
  387.     $handle = @opendir($sourcedir);
  388.     while ($eintrag = @readdir($handle)) {
  389.         $target = $sourcedir.$eintrag;
  390.         if (is_dir($target) && $eintrag != "." && $eintrag != "..") {
  391.             @unlink($target);
  392.             $slashpos = strrpos($target, "/");
  393.             if ($slashpos) {
  394.                 $deldir = substr($target, 0, $slashpos);
  395.                 @rmdir($deldir);
  396.             }
  397.             $filearray[$target] = DeleteDirectory($target);
  398.             @unlink($target);
  399.             $slashpos = strrpos($target, "/");
  400.             if ($slashpos) {
  401.                 $deldir = substr($target, 0, $slashpos);
  402.                 @rmdir($deldir);
  403.             }
  404.         } elseif ($eintrag != "." && $eintrag != "..") {
  405.             $filearray[] = $eintrag;
  406.             @unlink($target);
  407.             $slashpos = strrpos($target, "/");
  408.             if ($slashpos) {
  409.                 $deldir = substr($target, 0, $slashpos);
  410.                 @rmdir($deldir);
  411.             }
  412.         }
  413.     }
  414.     @closedir($handle);
  415.     return $filearray;
  416. }

  427.  * @return         array

  431. function apcms_ChmodDirectory($sourcedir, $mode=777) {
  432.     
  433.     $filemode = '0'.$mode;
  434.     $filemode = octdec($filemode);
  435.     
  436.     if (!is_dir($sourcedir)) {
  437.         return chmod($sourcedir, $filemode);
  438.     }
  439.     
  440.     $filearray = array();
  441.     $sourcedir .= "/";
  442.     $handle = @opendir($sourcedir);
  443.     while ($eintrag = @readdir($handle)) {
  444.         $target = $sourcedir.$eintrag;
  445.         if (is_dir($target) && $eintrag != "." && $eintrag != "..") {
  446.             @chmod($target, $filemode);
  447.             $slashpos = strrpos($target, "/");
  448.             if ($slashpos) {
  449.                 $chmoddir = substr($target, 0, $slashpos);
  450.                 @chmod($chmoddir, $filemode);
  451.             }
  452.             $filearray[$target] = apcms_ChmodDirectory($target, $mode);
  453.             @chmod($target, $filemode);
  454.             $slashpos = strrpos($target, "/");
  455.             if ($slashpos) {
  456.                 $chmoddir = substr($target, 0, $slashpos);
  457.                 @chmod($chmoddir, $filemode);
  458.             }
  459.         } elseif ($eintrag != "." && $eintrag != "..") {
  460.             $filearray[] = $eintrag;
  461.             @chmod($target, $filemode);
  462.             $slashpos = strrpos($target, "/");
  463.             if ($slashpos) {
  464.                 $chmoddir = substr($target, 0, $slashpos);
  465.                 @chmod($chmoddir, $filemode);
  466.             }
  467.         }
  468.     }
  469.     @closedir($handle);
  470.     return $filearray;
  471. }
