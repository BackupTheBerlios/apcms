  298.                 }
  299.                 $authorname = "";
  300.                 if ($this->config['show_author'] === true) {
  301.                     $retuser = $db->unbuffered_query_first("SELECT `nickname` FROM `".$apcms['table']['global']['users']."` WHERE `id`='".intval($retnews[$a][1])."'");

  306.                     }
  307.                 }
  308.                 
  309.                 $retcnum = $db->unbuffered_query_first("SELECT COUNT(*) FROM `".$this->commentstable."` WHERE `nid`='".intval($retnews[$a][0])."'");
